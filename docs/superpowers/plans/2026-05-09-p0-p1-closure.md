type StrictDogfoodAcceptance =
  | {

    };

const strictAcceptanceFromPreflight = (

async saveTraceEvent(traceEvent: TraceEventRecord): Promise<void> {
  const existing = this.traceEvents.get(traceEvent.id);
  this.traceEvents.set(
    traceEvent.id,
    clone({
      ...traceEvent,
      visibility: traceEvent.visibility ?? 'public',
      created_at: existing?.created_at ?? traceEvent.created_at,
    }),
  );
}

async saveTraceArtifactRef(traceArtifactRef: TraceArtifactRefRecord): Promise<void> {
  if (!this.traceArtifactRefs.has(traceArtifactRef.id)) {
    this.traceArtifactRefs.set(
      traceArtifactRef.id,
      clone({
        ...traceArtifactRef,
        ref_kind: traceArtifactRef.ref_kind ?? traceArtifactRef.ref.kind,
        redacted: traceArtifactRef.redacted ?? false,
      }),
    );
  }
}
